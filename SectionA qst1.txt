import java.util.Scanner;

public class Main
{
	public static void main(String[] args) {
		int n=7;
		Scanner sc= new Scanner(System.in);
		String s=sc.nextLine();
		for (int i=0;i<n ;i++ ) 
	    {
	        for (int j=0;j<n;j++ )
	        {
	            if(i+j==n/2||j-i==n/2||i-j==n/2||i+j==(n+(n/2)-1))
	               System.out.print(s.charAt(j));
	               else
	                System.out.print(" ");
	               
	            }
	       
	        System.out.println();
	       
	    }
	
	        
	        
	       
	    }
	}
